2.Greatest among 3 numbers
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
SQL>>DECLARE
    
	a integer:=100;
	
	b integer:=20;
	
	c integer:=130;
            
BEGIN
    if (a>b And a>c) then
			
		dbms_output.put_line('A is maximum among 3 and its value is '||a);
	
	         elsif(b>a And b>c) then
        
		dbms_output.put_line('B is maximum among 3 and its value is ' ||b);
	
	         else
        
		dbms_output.put_line('C is maximum among 3 and its value is '||c);
        
    
	         end if;    
            
END;

----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
OUTPUT:
Statement processed.
C is maximum among 3 and its value is 130

3.Grade calculation based on 3 numbers
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
SQL>DECLARE
   
 	English integer:=100;
	
	Maths integer:=100;
	
	Science integer:=100;
	
	m1 integer:=0;

          BEGIN
    
	m1:=((English+Maths+Science)/3);
    
	if (m1>=90 And m1<=100) then
		
		dbms_output.put_line('Students grade is A');
	
	elsif(m1>=80 And m1<90) then
        
		dbms_output.put_line('Students grade is B');
	
	elsif(m1>=70 And m1<80) then
        
		dbms_output.put_line('Students grade is C');
	
	elsif(m1>=60 And m1<70) then
        
		dbms_output.put_line('Students grade is D');
	
	elsif(m1>=50 And m1<60) then
        
		dbms_output.put_line('Students grade is E');
	
	else
        
		dbms_output.put_line('Student got F as grade and he/she is Failed in this exam');
        
    
	end if;    

          END;

----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
OUTPUT:
Statement processed.
Students grade is A

5.Reverse of a number
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
SQL>DECLARE
    
	Num integer:=4567;
	
	Rev integer:=0;
          
BEGIN
    
	While Num>0 loop
    	
	Rev:=(Rev*10) + mod(Num,10);
		
	Num:=floor(Num/10);

	END loop;
	
	dbms_output.put_line('Reverse of the number is: ' || Rev);

          END;


----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
OUTPUT:
Statement processed.
Reverse of the number is: 7654

6.Check for prime number
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
SQL>DECLARE
    
	Num integer:=8;
	
	flag integer:=0;

          BEGIN
    
	for i in 2..(Num/2) loop   	
   		 
	if((mod(Num, i))=0)then
    		
		flag:=1;
    		
		EXIT;
		 
	end if;
    
	end loop;

          if (flag=0) then
    
	dbms_output.put_line('Given number is a prime number');

          else
    
	dbms_output.put_line('Given number is not a prime number');

          END if;


END;
----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------


OUTPUT:
Statement processed.
Given number is not a prime number
